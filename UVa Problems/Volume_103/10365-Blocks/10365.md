# 題解

> [!warning]
> - 題目有說必須結合成長方形，也就是 $N = i \times j \times k$

暴力窮舉所有表面積情況即可。
# Code

```cpp
#include <iostream>
#include <climits>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false); cin.tie(nullptr);
	
	int C;  cin >> C;
	while (C--)
	{
		int N;  cin >> N;
		
		int ans = INT_MAX;
		for (int i = 1; i <= N; i++)
		{
			if (N % i == 0)
			{
				for (int j = 1; j * i <= N; j++)
				{
					if (N % (i * j) == 0)
					{
						for (int k = 1; j * i * k <= N; k++)
						{
							if (N == (i * j * k))
							{
								ans = min(ans, i*j*2 + i*k*2 + j*k*2);
							}
						}
					}
				}
			}
		}
		
		cout << ans << '\n';
	}
	
	return 0;
}
```
