# Code

```cpp
#include <iostream>
#include <string>
#include <cctype>
using namespace std;

int main()
{
    // 完整的鍵盤字元（小寫）
    string s = "`1234567890-=qwertyuiop[]\\asdfghjkl;'zxcvbnm,./";
    char c;
    
    while (cin.get(c))  // 一次讀一個字元
    {
        // 如果是空格或換行，直接輸出
        if (c == ' ' || c == '\n') 
        {
        	cout.put(c);
        	continue;
        }
        
        for (int i = 2; i < (int)s.length(); ++i)  // 從位置2開始找
        {
            if (s[i] == c)
            {    
                cout.put(s[i-2]);
                break;
            } 
    	} 
    }
    
    return 0;
}
```
