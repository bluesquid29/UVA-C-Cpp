# Code

## Use `<regex>`

```cpp
#include <iostream>
#include <string>
#include <regex>
using namespace std;

int main() 
{
	string noun     = "(tom|jerry|goofy|mickey|jimmy|dog|cat|mouse)";
	string article  = "(a|the)";
	string verb     = "(hate|love|know|like)s?";          // s? = 0或1個s
	
	string actor    = "((" + article + " +)?" + noun + ")";
	//                  └─ ARTICLE?  ─┘        └─NOUN─┘
	
	string act_list = "(" + actor + "( +and +" + actor + ")*)";
	//                  └─ ACTOR ─┘  └────── (and ACTOR)* ────┘
	
	string action   = "(" + act_list + " +" + verb + " +" + act_list + ")";
	//                  └─ACTIVE_LIST─┘      └VERB┘       └─ACTIVE_LIST─┘
	
	string statement= "^ *" + action + "( *, *" + action + ")* *$";
	//                          └─ACTION─┘  └────(, ACTION)*────┘
    
    regex regEx(statement);
    string s;
    
    while (getline(cin, s)) 
    {
        if (regex_match(s, regEx))  { cout << "YES I WILL\n"; }
        else                        { cout << "NO I WON'T\n"; }
    }
    
    return 0;
}
```
