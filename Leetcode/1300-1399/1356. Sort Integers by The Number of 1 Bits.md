# Code

```cpp
#include <print>
#include <vector>
#include <bit>
using namespace std;

class Solution {
public:
    vector<int> sortByBits(vector<int>& arr) {
        sort(arr.begin(), arr.end(), [](unsigned const& a, unsigned const& b)
        {
        	unsigned cnt_A = popcount(a);
        	unsigned cnt_B = popcount(b);
            return cnt_A == cnt_B ? a < b : cnt_A < cnt_B;
        });
        
        return arr;
    }
};

int main()
{
	Solution sol;
	
	vector<int> v = {1024, 512, 256, 128, 64, 32, 16, 8, 4, 2, 1};
	println("{}", sol.sortByBits(v));
	
	return 0;
}
```
