| A   | B   | XOR |     |
| --- | --- | --- | --- |
| 0   | 1   |     | 1   |
| 1   | 1   |     | 0   |

# Code

```cpp
class Solution {
public:
    string findDifferentBinaryString(vector<string>& nums)
    {
        int n = nums.size();
        string ans(n, 'X');

        for (int i = 0; i < n; i++)
        {
            ans[i] = nums[i][i] ^ 1;
        }

        return ans;
    }
};
```
