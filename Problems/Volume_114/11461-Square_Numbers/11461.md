# Code

```cpp
#include <iostream>    
#include <cmath>       // sqrt, floor, ceil
using namespace std;

int main()
{	
	int a, b;
	while (cin >> a >> b)
	{
		if (0 == a && 0 == b) break;	
		cout << floor(sqrt(b)) - ceil(sqrt(a)) + 1 << '\n';
	}
	
	return 0;
}
```
