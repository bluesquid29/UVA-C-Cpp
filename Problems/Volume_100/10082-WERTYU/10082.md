# Code

```cpp
# include <cstdio>

int main()
{
	char s[] = "`1234567890-=QWERTYUIOP[]\\ASDFGHJKL;'ZXCVBNM,./";
	
	char c;
	int i;
	
	while ((c = getchar()) != EOF)
	{
		if (c == ' ' || c == '\n')
		{
			putchar(c);
		}
		else
		{
			for (i = 1; s[i] != '\0'; i++)
			{
				if (s[i] == c)
				{	
					putchar(s[i-1]);
					break;
				} 
			}
		}	
	}
	
	return 0;
}
```
